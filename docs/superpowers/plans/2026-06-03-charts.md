**Goal:** Add a `charts` subcommand to agent-bencher that reads existing `batch.json` files and generates comparative bar/line chart PNGs for internal analysis.

**Architecture:** A single `charts.py` module with matplotlib-based chart generation, integrated as a new CLI subcommand in the existing `cli.py`. The module reads from `runs/<conversation>/<agent-id>/<batch-id>/batch.json` files, extracts `run_metrics` and `turn_metrics`, and produces side-by-side bar charts with error bars (min/max range) and per-turn line charts.

**Tech Stack:** Python 3.12+, matplotlib (optional dependency), existing agent-bencher infrastructure.

from __future__ import annotations

from pathlib import Path
import json


def _load_batch_data(batch_dir: Path) -> dict | None:
    """Load and validate a batch.json file.

    Returns None if the file doesn't exist or is invalid.
    """
    batch_path = batch_dir / "batch.json"
    if not batch_path.is_file():
        return None
    try:
        data = json.loads(batch_path.read_text())
    except (json.JSONDecodeError, OSError):
        return None
    if "run_metrics" not in data or "turn_metrics" not in data:
        return None
    return data
```

import math


def _plot_grouped_bar_chart(
    all_batches: list[dict],
    metric_name: str,
    title: str,
    y_label: str,
    output_path: Path,
) -> Path:
    """Generate a side-by-side bar chart with error bars (min/max range).

    Each batch represents one agent. The bar height is the mean,
    and error bars span from min to max.
    """
    import matplotlib
    matplotlib.use("Agg")
    import matplotlib.pyplot as plt
    import numpy as np

    if not all_batches:
        output_path.parent.mkdir(parents=True, exist_ok=True)
        output_path.touch()
        return output_path

    agent_ids = [b["agent_id"] for b in all_batches]
    means = [b["run_metrics"][metric_name]["mean"] for b in all_batches]
    mins = [b["run_metrics"][metric_name]["min"] for b in all_batches]
    maxs = [b["run_metrics"][metric_name]["max"] for b in all_batches]

    # Error bars: distance from mean to min (down) and max (up)
    error_down = [means[i] - mins[i] for i in range(len(means))]
    error_up = [maxs[i] - means[i] for i in range(len(means))]

    fig, ax = plt.subplots(figsize=(max(8, len(agent_ids) * 1.5), 5))
    x = np.arange(len(agent_ids))
    width = 0.6

    ax.bar(x, means, width, yerr=[error_down, error_up], capsize=4, color="#4a90d9", label="mean (min-max range)")

    ax.set_xticks(x)
    ax.set_xticklabels(agent_ids, rotation=30, ha="right")
    ax.set_title(title)
    ax.set_ylabel(y_label)
    ax.legend()

    fig.tight_layout()
    output_path.parent.mkdir(parents=True, exist_ok=True)
    fig.savefig(str(output_path), dpi=150)
    plt.close(fig)

    return output_path
```

def _plot_per_turn_line_chart(
    all_batches: list[dict],
    metric_name: str,
    title: str,
    y_label: str,
    output_path: Path,
) -> Path:
    """Generate a per-turn line chart showing how each agent performs across turns.

    X-axis is turn number. Each agent gets a line with error bars (min/max range).
    """
    import matplotlib
    matplotlib.use("Agg")
    import matplotlib.pyplot as plt

    if not all_batches:
        output_path.parent.mkdir(parents=True, exist_ok=True)
        output_path.touch()
        return output_path

    # Determine number of turns from the first batch with turn_metrics
    turn_count = 0
    for b in all_batches:
        if b.get("turn_metrics"):
            turn_count = len(b["turn_metrics"])
            break

    if turn_count == 0:
        output_path.parent.mkdir(parents=True, exist_ok=True)
        output_path.touch()
        return output_path

    fig, ax = plt.subplots(figsize=(max(8, turn_count * 1.5), 5))

    colors = ["#4a90d9", "#e74c3c", "#2ecc71", "#f39c12", "#9b59b6", "#1abc9c", "#e67e22", "#34495e"]

    for idx, batch in enumerate(all_batches):
        agent_id = batch["agent_id"]
        turn_metrics = batch.get("turn_metrics", [])
        if not turn_metrics:
            continue

        means = [turn_metrics[t][metric_name]["mean"] for t in range(turn_count)]
        mins = [turn_metrics[t][metric_name]["min"] for t in range(turn_count)]
        maxs = [turn_metrics[t][metric_name]["max"] for t in range(turn_count)]

        error_down = [means[t] - mins[t] for t in range(turn_count)]
        error_up = [maxs[t] - means[t] for t in range(turn_count)]

        color = colors[idx % len(colors)]
        ax.errorbar(
            range(turn_count),
            means,
            yerr=[error_down, error_up],
            capsize=3,
            color=color,
            label=agent_id,
            marker="o",
            linewidth=1.5,
        )

    ax.set_xticks(range(turn_count))
    ax.set_xticklabels([f"Turn {t+1}" for t in range(turn_count)])
    ax.set_title(title)
    ax.set_ylabel(y_label)
    ax.set_xlabel("Turn")
    ax.legend()

    fig.tight_layout()
    output_path.parent.mkdir(parents=True, exist_ok=True)
    fig.savefig(str(output_path), dpi=150)
    plt.close(fig)

    return output_path
```

RUN_METRICS = [
    "duration_seconds",
    "total_input_tokens",
    "total_output_tokens",
    "effective_output_tps",
    "effective_total_throughput_tps",
]

TURN_METRICS = [
    "duration_seconds",
    "output_tps",
]

_METRIC_LABELS = {
    "duration_seconds": "Duration (s)",
    "total_input_tokens": "Input tokens",
    "total_output_tokens": "Output tokens",
    "effective_output_tps": "Output TPS",
    "effective_total_throughput_tps": "Total throughput TPS",
}


def generate_charts(conversation_dir: Path, output_dir: Path | None = None) -> list[Path]:
    """Generate comparative charts from all batch.json files under a conversation directory.

    Scans <conversation_dir>/<agent-id>/<batch-id>/batch.json for all agents and batches.
    Generates run-level bar charts and per-turn line charts.

    Args:
        conversation_dir: Path to a directory like runs/<conversation-name>/ containing
                          agent subdirectories, each with batch directories containing batch.json.
        output_dir: Where to write PNG files. Defaults to <conversation_dir>/charts/.

    Returns:
        List of paths to generated chart files.
    """
    if output_dir is None:
        output_dir = conversation_dir / "charts"

    output_dir.mkdir(parents=True, exist_ok=True)
    generated: list[Path] = []

    # Find all batch.json files
    all_batches: list[dict] = []
    agent_ids: list[str] = []

    if not conversation_dir.is_dir():
        return generated

    for agent_dir in sorted(conversation_dir.iterdir()):
        if not agent_dir.is_dir():
            continue
        for batch_dir in sorted(agent_dir.iterdir()):
            data = _load_batch_data(batch_dir)
            if data is None:
                continue
            all_batches.append(data)
            if data["agent_id"] not in agent_ids:
                agent_ids.append(data["agent_id"])

    if not all_batches:
        return generated

    # Generate run-level bar charts (one per metric, all agents side-by-side)
    for metric_name in RUN_METRICS:
        # Check if any batch has this metric
        has_metric = any(metric_name in b.get("run_metrics", {}) for b in all_batches)
        if not has_metric:
            continue
        chart_path = output_dir / f"{metric_name}.png"
        _plot_grouped_bar_chart(
            all_batches=all_batches,
            metric_name=metric_name,
            title=f"{_METRIC_LABELS.get(metric_name, metric_name)} — All Agents",
            y_label=_METRIC_LABELS.get(metric_name, metric_name),
            output_path=chart_path,
        )
        generated.append(chart_path)

    # Generate per-turn line charts
    per_turn_dir = output_dir / "per_turn"
    for metric_name in TURN_METRICS:
        # Check if any batch has this metric in turn_metrics
        has_metric = any(
            metric_name in t for b in all_batches for t in b.get("turn_metrics", [])
        )
        if not has_metric:
            continue
        chart_path = per_turn_dir / f"{metric_name}.png"
        _plot_per_turn_line_chart(
            all_batches=all_batches,
            metric_name=metric_name,
            title=f"{_METRIC_LABELS.get(metric_name, metric_name)} — By Turn",
            y_label=_METRIC_LABELS.get(metric_name, metric_name),
            output_path=chart_path,
        )
        generated.append(chart_path)

    return generated
```

def _cmd_charts(args: argparse.Namespace) -> int:
    """Generate comparative charts from benchmark runs."""
    from agent_bencher.charts import generate_charts

    input_path = args.input

    # Resolve input: could be a YAML file or a runs/ directory
    if input_path.suffix == ".yaml" or input_path.suffix == ".yml":
        # Resolve conversation name from YAML
        import yaml
        data = yaml.safe_load(input_path.read_text())
        conversation_name = data.get("name", input_path.stem)
        conversation_dir = Path("runs") / conversation_name
    else:
        conversation_dir = input_path

    if not conversation_dir.is_dir():
        print(f"Error: directory not found: {conversation_dir}", file=sys.stderr)
        return 1

    output_dir = args.output_dir
    if output_dir is not None:
        output_dir = output_dir

    generated = generate_charts(conversation_dir, output_dir=output_dir)
    print(f"Generated {len(generated)} chart(s) to {output_dir or conversation_dir / 'charts'}", file=sys.stderr)
    for path in generated:
        print(f"  {path}", file=sys.stderr)
    return 0
